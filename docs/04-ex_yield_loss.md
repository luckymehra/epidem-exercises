### Transform dataset a


```r
a_yield <- a %>% 
	dplyr::select(BLK, TRT, YIELD) %>% 
	arrange(BLK, TRT, YIELD) %>% 
	group_by(BLK, TRT) %>% 
	summarise(RY2 = YIELD[1]/YIELD[2]) %>% 
	ungroup()


a_ds <- a %>% 
	dplyr::select(BLK, TRT, DS) %>% 
	arrange(BLK, TRT, DS) %>% 
	group_by(BLK, TRT) %>% 
	summarise(CDS = DS[2]) %>% 
	ungroup()

a_new <- a_yield %>% 
	inner_join(a_ds) %>% 
	ungroup() %>% 
	mutate(BLK = parse_factor(as.character(BLK)),
				 TRT = parse_factor(as.character(TRT)))
```
